# Doll_Detection_YOLO
This project uses Roboflow to train a personalized YOLO class

# Create environment
python -m venv env_doll

# Activate environment
env_doll\Scripts\activate     # (Windows)
# or
source env_doll/bin/activate  # (Mac/Linux)

# Install dependencies
pip install -r requirements.txt

Finally run 
python Doll_detection.py
